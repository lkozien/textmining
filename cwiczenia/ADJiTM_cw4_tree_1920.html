Przygotowanie znormalizowanego zbioru danych

Normalizacja kadej zmiennej z wyczeniem atrybutu decyzyjnego:

normalizuj <- function(x){
  (x-min(x))/(max(x)-min(x))
}

spam_norm <- spam %>% 
  select(-type) %>% 
  mutate_all(normalizuj)

spam_norm <-  spam_norm %>% 
  bind_cols(type = spam$type)

  select(-type) %>%

rec_korpus <- tm_map(rec_korpus, removeWords, c(stopwords("en"), "one", "two", "three", "make", "get", "movie", "movies", "film", "films"))
rec_korpus <- tm_map(rec_korpus, stemDocument)
